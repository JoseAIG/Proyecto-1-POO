//TABLA ANIMALES 
create table animales (identificacion integer not null, animal varchar(10), especie varchar(10), apodo varchar(15), edad integer not null, peso numeric(7,2), primary key (identificacion));

//TABLA AGUILAS
create table aguilas (identificacion integer not null, apodo varchar(15), edad integer not null, peso numeric(7,2), primary key(identificacion), foreign key(identificacion) references animales on update cascade on delete no action);

//TABLA PINGUINOS
create table pinguinos (identificacion integer not null, apodo varchar(15), edad integer not null, peso numeric(7,2), primary key(identificacion), foreign key(identificacion) references animales on update cascade on delete no action);

//TABLA SERPIENTES
create table serpientes (identificacion integer not null, apodo varchar(15), edad integer not null, peso numeric(7,2), tamanio varchar(15),primary key(identificacion), foreign key(identificacion) references animales on update cascade on delete no action);

//TABLA COCODRILOS

//TABLA GUEPARDOS

//TABLA PUMAS

//TABLA GORILAS
create table gorilas (identificacion integer not null, apodo varchar(15), edad integer not null, peso numeric(7,2), primary key(identificacion), foreign key(identificacion) references animales on update cascade on delete no action);
